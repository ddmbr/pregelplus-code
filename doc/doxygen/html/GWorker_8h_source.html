163  void agg_sync()
164  {
165  AggregatorT* agg = (AggregatorT*)get_aggregator();
166  if (agg != NULL) {
167  if (_my_rank != MASTER_RANK) { //send partialT to aggregator
168  //gathering PartialT
169  PartialT* part = agg->finishPartial();
170  //------------------------ strategy choosing BEGIN ------------------------
171  StartTimer(COMMUNICATION_TIMER);
172  StartTimer(SERIALIZATION_TIMER);
173  ibinstream m;
174  m << part;
175  int sendcount = m.size();
176  StopTimer(SERIALIZATION_TIMER);
177  int total = all_sum(sendcount);
178  StopTimer(COMMUNICATION_TIMER);
179  //------------------------ strategy choosing END ------------------------
180  if (total <= AGGSWITCH)
181  slaveGather(*part);
182  else {
183  send_ibinstream(m, MASTER_RANK);
184  }
185  //scattering FinalT
186  slaveBcast(*((FinalT*)global_agg));
187  } else {
188  //------------------------ strategy choosing BEGIN ------------------------
189  int total = all_sum(0);
190  //------------------------ strategy choosing END ------------------------
191  //gathering PartialT
192  if (total <= AGGSWITCH) {
193  vector<PartialT*> parts(_num_workers);
194  masterGather(parts);
195  for (int i = 0; i < _num_workers; i++) {
196  if (i != MASTER_RANK) {
197  PartialT* part = parts[i];
198  agg->stepFinal(part);
199  delete part;
200  }
201  }
202  } else {
203  for (int i = 0; i < _num_workers; i++) {
204  if (i != MASTER_RANK) {
205  obinstream um = recv_obinstream(i);
206  PartialT* part;
207  um >> part;
208  agg->stepFinal(part);
209  delete part;
210  }
211  }
212  }
213  //scattering FinalT
214  FinalT* final = agg->finishFinal();
215  //cannot set "global_agg=final" since MASTER_RANK works as a slave, and agg->finishFinal() may change
216  *((FinalT*)global_agg) = *final; //deep copy
217  masterBcast(*((FinalT*)global_agg));
218  }
219  }

351  global_step_num++;
352  ResetTimer(4);
353  //===================
354  char bits_bor = all_bor(global_bor_bitmap);
355  if (getBit(FORCE_TERMINATE_ORBIT, bits_bor) == 1)
356  break;
357  get_vnum() = all_sum(vertexes.size());
358  int wakeAll = getBit(WAKE_ALL_ORBIT, bits_bor);
359  if (wakeAll == 0) {
360  active_vnum() = all_sum(active_count);
361  if (active_vnum() == 0 && getBit(HAS_MSG_ORBIT, bits_bor) == 0)
362  break; //all_halt AND no_msg
363  } else
364  active_vnum() = get_vnum();
365  //===================
366  AggregatorT* agg = (AggregatorT*)get_aggregator();
367  if (agg != NULL)

369  //===================
370  clearBits();
371  if (wakeAll == 1)
372  all_compute();
373  else
374  active_compute();

377  step_vadd_num = master_sum_LL(message_buffer->get_total_vadd());
378  step_gmsg_num = master_sum_LL(message_buffer->get_total_gmsg());
379  if (_my_rank == MASTER_RANK) {
380  global_msg_num += step_msg_num;
381  global_vadd_num += step_vadd_num;
382  global_gmsg_num += step_gmsg_num;
383  }
384  vector<GVertexT*>& to_add = message_buffer->sync_messages(ghosts);
385  agg_sync();
386  for (int i = 0; i < to_add.size(); i++)
387  add_vertex(to_add[i]);
388  to_add.clear();
389  //===================
390  worker_barrier();
391  StopTimer(4);
392  if (_my_rank == MASTER_RANK) {
393  cout << "Superstep " << global_step_num << " done. Time elapsed: " << get_timer(4) << " seconds" << endl;
394  cout << "#msgs: " << step_msg_num << ", #vadd: " << step_vadd_num << ", #gmsg: " << step_gmsg_num << endl;
395  }
396  }
397  worker_barrier();
398  StopTimer(WORKER_TIMER);
399  PrintTimer("Communication Time", COMMUNICATION_TIMER);
400  PrintTimer("- Serialization Time", SERIALIZATION_TIMER);
401  PrintTimer("- Transfer Time", TRANSFER_TIMER);
402  PrintTimer("Total Computational Time", WORKER_TIMER);
403  if (_my_rank == MASTER_RANK)
404  cout << "Total #msgs=" << global_msg_num << ", Total #vadd=" << global_vadd_num << ", Total #gmsg=" << global_gmsg_num << endl;
405 
406  // dump graph

409  worker_barrier();

412  }
